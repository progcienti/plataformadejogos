procedimento carrega_sorteada()
var i: inteiro
inicio

   para i  de 1 ate tamanho faca
      sorteada[i] <- copia(palavras[sorteio], i,1) // Funcao copia retorna a letra na posicao informada
      // retorna
   fimpara

   para i de 1 ate tamanho faca
      jogo[i]<-" "
   fimpara

fimprocedimento

procedimento verifica_letra()
var i: inteiro
   achou: logico
inicio
   achou <- falso
   para i de 1 ate tamanho faca
      se (letra = sorteada[i]) entao
         jogo[i] <- letra
         achou <- verdadeiro
         cont<-cont+1
      fimse
   fimpara
   se achou = falso entao
      erros <- erros + 1
   fimse

procedimento posicionarFruta()
inicio
   frutaX <- randi(TAM - 2) + 2   // Não gerar na parede de cima
   frutaY <- randi(TAM - 2) + 2   // Não gerar nas laterais
   enquanto campo1[frutaX, frutaY] <> " " faca
      frutaX <- randi(TAM - 2) + 2
      frutaY <- randi(TAM - 2) + 2
   fimenquanto
   campo1[frutaX, frutaY] <- "F"
